⚡ DIVIDE & CONQUER
Binary Search
Search a sorted array by halving the search space each step. O(log n) — way faster than linear search.
How It Works
Array must be sorted. Compare target with middle element. If smaller → search left half. If larger → search right half. Array: [2, 5, 8, 12, 16, 23, 38, 56, 72, 91] Target = 23 Step 1: low=0 mid=4 high=9 → arr[4]=16 < 23 → go RIGHT Step 2: low=5 mid=7 high=9 → arr[7]=56 > 23 → go LEFT Step 3: low=5 mid=5 high=6 → arr[5]=23 ✓ FOUND at index 5!
Algorithm
PSEUDOCODE
BinarySearch(arr, low, high, key)
{
  while (low ≤ high) do
  {
    mid := (low + high) / 2;
    if (arr[mid] = key) then return mid;
    else if (arr[mid] < key) then low := mid + 1;
    else high := mid - 1;
  }
  return -1;  // Not found
}
Complexity
Best
O(1)
Average
O(log n)
Worst
O(log n)

import java.util.Scanner;

public class BinarySearch {

    static int binarySearch(int[] arr, int key) {
        int low = 0, high = arr.length - 1;
        while (low <= high) {
            int mid = (low + high) / 2;
            if (arr[mid] == key)       return mid;
            else if (arr[mid] < key)   low = mid + 1;
            else                        high = mid - 1;
        }
        return -1;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter number of elements: ");
        int n = sc.nextInt();
        int[] arr = new int[n];
        System.out.println("Enter sorted elements:");
        for (int i = 0; i < n; i++) arr[i] = sc.nextInt();
        System.out.print("Enter key to search: ");
        int key = sc.nextInt();
        int result = binarySearch(arr, key);
        if (result != -1)
            System.out.println("Found at index: " + result);
        else
            System.out.println("Element not found");
    }
}

⚡ DIVIDE & CONQUER
Quick Sort
Pick a pivot, place smaller elements left and larger right. Pivot is now in final position. Recurse on both halves.
Partition Step
Array: [10, 80, 30, 90, 40, 50, 70] Pivot = 70 (last) i = -1 j=0: 10 < 70 → i=0 → [10, 80, 30, 90, 40, 50, 70] j=2: 30 < 70 → i=1 → [10, 30, 80, 90, 40, 50, 70] j=4: 40 < 70 → i=2 → [10, 30, 40, 90, 80, 50, 70] j=5: 50 < 70 → i=3 → [10, 30, 40, 50, 80, 90, 70] swap arr[4] with pivot → [10,30,40,50 70 90,80] ✓
Algorithm
PSEUDOCODE
QuickSort(arr, low, high)
{
  if (low < high) then
  {
    pi := Partition(arr, low, high);
    QuickSort(arr, low, pi-1);
    QuickSort(arr, pi+1, high);
  }
}

Partition(arr, low, high)
{
  pivot := arr[high];  i := low - 1;
  for j := low to high-1 do
    if (arr[j] ≤ pivot) then { i++; swap(arr[i], arr[j]); }
  swap(arr[i+1], arr[high]);
  return i+1;
}
Complexity
Best
O(n log n)
Average
O(n log n)
Worst (sorted)

import java.util.Scanner;

public class QuickSort {

    static int partition(int[] arr, int low, int high) {
        int pivot = arr[high], i = low - 1;
        for (int j = low; j < high; j++) {
            if (arr[j] <= pivot) {
                i++;
                int tmp = arr[i]; arr[i] = arr[j]; arr[j] = tmp;
            }
        }
        int tmp = arr[i+1]; arr[i+1] = arr[high]; arr[high] = tmp;
        return i + 1;
    }

    static void quickSort(int[] arr, int low, int high) {
        if (low < high) {
            int pi = partition(arr, low, high);
            quickSort(arr, low, pi - 1);
            quickSort(arr, pi + 1, high);
        }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter number of elements: ");
        int n = sc.nextInt();
        int[] arr = new int[n];
        System.out.println("Enter elements:");
        for (int i = 0; i < n; i++) arr[i] = sc.nextInt();
        quickSort(arr, 0, n - 1);
        System.out.print("Sorted: ");
        for (int x : arr) System.out.print(x + " ");
    }
}

⚡ DIVIDE & CONQUER
Merge Sort
Divide array in half recursively until single elements, then merge sorted halves back together.
Visualization
Original: [38 27 43 3 9 82 10] Split: [38 27 43] [3 9 82 10] [38][27 43] [3 9][82 10] Merge up: [27 38]→[27 38 43] [3 9]→[3 9 10 82] Final: [3 9 10 27 38 43 82] ✓
Algorithm
PSEUDOCODE
MergeSort(arr, low, high)
{
  if (low < high) then
  {
    mid := (low + high) / 2;
    MergeSort(arr, low, mid);
    MergeSort(arr, mid+1, high);
    Merge(arr, low, mid, high);
  }
}

Merge(arr, l, m, r)
{
  Create L[0..n1], R[0..n2];
  Copy arr[l..m] → L,  arr[m+1..r] → R;
  i := 0; j := 0; k := l;
  while (i < n1 and j < n2) do
    if (L[i] ≤ R[j]) then arr[k++] := L[i++]
    else arr[k++] := R[j++];
  Copy remaining elements;
}
Complexity
Best
O(n log n)
Average
O(n log n)
Worst

🌟 Always O(n log n) but needs O(n) extra space. Stable sort.

import java.util.Scanner;

public class MergeSort {

    static void merge(int[] arr, int l, int m, int r) {
        int n1 = m-l+1, n2 = r-m;
        int[] L = new int[n1], R = new int[n2];
        for (int i=0; i<n1; i++) L[i] = arr[l+i];
        for (int j=0; j<n2; j++) R[j] = arr[m+1+j];
        int i=0, j=0, k=l;
        while (i<n1 && j<n2)
            arr[k++] = (L[i]<=R[j]) ? L[i++] : R[j++];
        while (i<n1) arr[k++]=L[i++];
        while (j<n2) arr[k++]=R[j++];
    }

    static void mergeSort(int[] arr, int l, int r) {
        if (l < r) {
            int m = (l+r)/2;
            mergeSort(arr, l, m);
            mergeSort(arr, m+1, r);
            merge(arr, l, m, r);
        }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter number of elements: ");
        int n = sc.nextInt();
        int[] arr = new int[n];
        System.out.println("Enter elements:");
        for (int i=0; i<n; i++) arr[i] = sc.nextInt();
        mergeSort(arr, 0, n-1);
        System.out.print("Sorted: ");
        for (int x : arr) System.out.print(x + " ");
    }
}

🌐 GRAPH TRAVERSAL
BFS & DFS
BFS uses a Queue — level by level. DFS uses recursion/Stack — deep first. Both visit every node once: O(V+E).
BFS — uses Queue
1→[2,3], 2→[4,5], 3→[6] Q:[1] → visit 1 Q:[2,3] → visit 2 Q:[3,4,5] → visit 3 Q:[4,5,6] → done Order: 1 2 3 4 5 6
DFS — uses Recursion
Same graph visit 1 → recurse 2 visit 2 → recurse 4 visit 4 ← back visit 5 ← back visit 3 → visit 6 Order: 1 2 4 5 3 6
Algorithm
PSEUDOCODE
BFS(graph, start)
{
  Mark start as visited; Enqueue(Q, start);
  while (Q not empty) do
  {
    v := Dequeue(Q); print(v);
    for each w adjacent to v do
      if (w not visited) then
        { Mark w visited; Enqueue(Q, w); }
  }
}

DFS(graph, v)
{
  Mark v as visited; print(v);
  for each w adjacent to v do
    if (w not visited) then DFS(graph, w);
}

import java.util.*;

public class GraphTraversal {
    static int[][] adj;
    static boolean[] visited;
    static int n;

    static void bfs(int start) {
        Queue<Integer> queue = new LinkedList<>();
        visited[start] = true;
        queue.add(start);
        while (!queue.isEmpty()) {
            int v = queue.poll();
            System.out.print(v + " ");
            for (int i = 1; i <= n; i++)
                if (adj[v][i] == 1 && !visited[i]) {
                    visited[i] = true;
                    queue.add(i);
                }
        }
    }

    static void dfs(int v) {
        visited[v] = true;
        System.out.print(v + " ");
        for (int i = 1; i <= n; i++)
            if (adj[v][i] == 1 && !visited[i])
                dfs(i);
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter vertices and edges: ");
        n = sc.nextInt(); int e = sc.nextInt();
        adj = new int[n+1][n+1];
        visited = new boolean[n+1];
        for (int i=0; i<e; i++) {
            int u=sc.nextInt(), v=sc.nextInt();
            adj[u][v] = adj[v][u] = 1;
        }
        System.out.print("BFS: "); bfs(1);
        Arrays.fill(visited, false);
        System.out.print("\nDFS: "); dfs(1);
    }
}

🔗 WEEK 2 —
Articulation Points & Biconnected Components
An Articulation Point (AP) is a vertex whose removal disconnects the graph. Biconnected components are maximal subgraphs with no APs.
Algorithm Art(u, v)
MA'AM'S ALGORITHM
// u is start vertex for DFS. v is its parent.
// dfn[] initialized to 0, num initialized to 1.
Art(u, v)
{
  dfn[u] := num;  L[u] := num;  num := num + 1;
  for each vertex w adjacent from u do
  {
    if (dfn[w] = 0) then
    {
      Art(w, u);         // w is unvisited
      L[u] := min(L[u], L[w]);
    }
    else if (w ≠ v) then  L[u] := min(L[u], dfn[w]);
  }
}

import java.util.*;

class BiconnectedComponents {
    static final int MAX = 20;
    static int[] dfn = new int[MAX], l = new int[MAX];
    static int num = 1, n;
    static int[][] adjMatrix = new int[MAX][MAX];
    static Stack<Edge> stack = new Stack<>();

    static class Edge {
        int u, v;
        Edge(int u, int v) { this.u = u; this.v = v; }
    }

    static void bicomp(int u, int v) {
        dfn[u] = num;  l[u] = num;  num++;
        for (int w = 1; w <= n; w++) {
            if (adjMatrix[u][w] == 1) {
                if ((v != w) && (dfn[w] < dfn[u])) {
                    stack.push(new Edge(u, w));
                    System.out.println("Edge " + u + " -- " + w + " pushed.");
                }
                if (dfn[w] == 0) {
                    bicomp(w, u);
                    l[u] = Math.min(l[u], l[w]);
                    if (l[w] >= dfn[u]) {
                        System.out.println("Biconnected Component (AP=" + u + "):");
                        Edge e;
                        do {
                            e = stack.pop();
                            System.out.print(e.u + "--" + e.v + " ");
                        } while (!((e.u==u && e.v==w) || (e.u==w && e.v==u)));
                        System.out.println();
                    }
                } else if (w != v) {
                    l[u] = Math.min(l[u], dfn[w]);
                }
            }
        }
    }

🗺️ WEEK 4 —
Dijkstra's Algorithm
Shortest path from a source to all other vertices in a weighted graph. Greedy — always pick unvisited vertex with minimum distance.
Algorithm ShortestPaths
MA'AM'S ALGORITHM
// dist[j] = length of shortest path from v to j
ShortestPaths(v, cost, dist, n)
{
  for i := 1 to n do
  {  S[i] := false;  dist[i] := cost[v, i];  }
  S[v] := true;  dist[v] := 0.0;
  for num := 2 to n-1 do
  {
    Choose u: not in S, dist[u] is minimum;
    S[u] := true;
    for each w adjacent to u with S[w]=false do
      if (dist[w] > dist[u] + cost[u,w]) then
        dist[w] := dist[u] + cost[u,w];
  }
}
Complexity
Simple Array
O(V²)
Min Heap
O((V+E)logV)
Note
No -ve edges!

import java.util.Scanner;

public class Dijkstra {
    static final int MAX = 999;
    static final int MAX_N = 100;

    static int minDistance(double[] dist, boolean[] S, int n) {
        double min = MAX;  int minIndex = -1;
        for (int i = 1; i <= n; i++)
            if (!S[i] && dist[i] < min) { min = dist[i]; minIndex = i; }
        return minIndex;
    }

    static void shortestPaths(int v, double[][] cost, double[] dist, int n) {
        boolean[] S = new boolean[MAX_N];
        for (int i = 1; i <= n; i++) { S[i] = false; dist[i] = cost[v][i]; }
        S[v] = true;  dist[v] = 0.0;
        for (int num = 2; num <= n; num++) {
            int u = minDistance(dist, S, n);
            if (u == -1) break;
            S[u] = true;
            for (int w = 1; w <= n; w++)
                if (!S[w] && dist[w] > dist[u] + cost[u][w])
                    dist[w] = dist[u] + cost[u][w];
        }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter the number of vertices: ");
        int n = sc.nextInt();
        double[][] cost = new double[MAX_N][MAX_N];
        double[] dist  = new double[MAX_N];
        System.out.println("Enter adjacency cost matrix (999 = infinity):");
        for (int i=1; i<=n; i++)
            for (int j=1; j<=n; j++)
                cost[i][j] = sc.nextDouble();
        System.out.print("Enter source vertex: ");
        int src = sc.nextInt();
        shortestPaths(src, cost, dist, n);
        System.out.println("Vertex\tDistance from " + src);
        for (int i=1; i<=n; i++) {
            if (dist[i] == MAX) System.out.println(i + "\tINF");
            else System.out.printf("%d\t%.2f%n", i, dist[i]);
        }
        sc.close();
    }
}

🌲 WEEK 3 —
Prim's & Kruskal's Algorithm
Both find the Minimum Spanning Tree (MST). Prim grows one tree greedily. Kruskal sorts all edges and adds cheapest without forming a cycle.
Algorithm Prim
MA'AM'S ALGORITHM
Prim(E, cost, n, t)
{
  Let (k,l) = edge of minimum cost in E;
  mincost := cost[k,l];  t[1,1]:=k;  t[1,2]:=l;
  for i:=1 to n do
    if (cost[i,l] < cost[i,k]) then near[i]:=l
    else near[i]:=k;
  near[k] := near[l] := 0;
  for i:=2 to n-1 do
  {
    j := index where near[j]≠0 and cost[j,near[j]] is min;
    t[i,1]:=j;  t[i,2]:=near[j];
    mincost := mincost + cost[j,near[j]];
    near[j] := 0;
    for k:=1 to n do
      if ((near[k]≠0) and (cost[k,near[k]] > cost[k,j]))
        then near[k]:=j;
  }
  return mincost;
}

import java.util.Scanner;

public class PrimsAlgorithm {
    static final int INF = 99999;

    public static void prim(int[][] cost, int n) {
        int[][] t = new int[n-1][2];
        int[] near = new int[n+1];
        int mincost = 0, k = 0, l = 0, min = INF;
        for (int i=1; i<=n; i++)
            for (int j=1; j<=n; j++)
                if (cost[i][j] < min) { min=cost[i][j]; k=i; l=j; }
        mincost = cost[k][l];
        t[0][0] = k;  t[0][1] = l;
        for (int i=1; i<=n; i++)
            near[i] = (cost[i][l] < cost[i][k]) ? l : k;
        near[k] = near[l] = 0;
        for (int i=1; i<n-1; i++) {
            int j = 0, minVal = INF;
            for (int m=1; m<=n; m++)
                if (near[m]!=0 && cost[m][near[m]]<minVal)
                    { minVal=cost[m][near[m]]; j=m; }
            t[i][0]=j;  t[i][1]=near[j];
            mincost += cost[j][near[j]];
            near[j] = 0;
            for (int m=1; m<=n; m++)
                if (near[m]!=0 && cost[m][near[m]]>cost[m][j])
                    near[m]=j;
        }
        System.out.println("MST Edges:");
        for (int i=0; i<n-1; i++)
            System.out.println("(" + t[i][0] + ", " + t[i][1] + ")");
        System.out.println("Minimum Cost: " + mincost);
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[][] cost = new int[n+1][n+1];
        for (int i=1; i<=n; i++) for (int j=1; j<=n; j++) cost[i][j]=sc.nextInt();
        prim(cost, n);  sc.close();
    }
}

Algorithm Kruskal
MA'AM'S ALGORITHM
Kruskal(E, cost, n, t)
{
  Heapify the edge costs;
  for i:=1 to n do parent[i] := -1;
  i:=0;  mincost:=0;
  while ((i < n-1) and (heap not empty)) do
  {
    Delete min edge (u,v) from heap;  Adjust heap;
    j := Find(u);  k := Find(v);
    if (j ≠ k) then
    {
      i++;  t[i,1]:=u;  t[i,2]:=v;
      mincost := mincost + cost[u,v];
      Union(j, k);
    }
  }
  if (i ≠ n-1) then write("No spanning tree");
  else return mincost;
}
Kruskal's Java Code
Java
import java.util.*;

public class KruskalAlgorithm {
    static int[] parent;
    static int n, heapSize;

    static class Edge {
        int u, v, cost;
        Edge(int u, int v, int c) { this.u=u; this.v=v; cost=c; }
    }
    static Edge[] heap;

    static void adjust(int i) {
        int j = 2*i;  Edge item = heap[i];
        while (j <= heapSize) {
            if (j < heapSize && heap[j].cost > heap[j+1].cost) j++;
            if (item.cost <= heap[j].cost) break;
            heap[j/2] = heap[j];  j *= 2;
        }
        heap[j/2] = item;
    }
    static void buildHeap() { for (int i=heapSize/2; i>=1; i--) adjust(i); }
    static Edge deleteMin() {
        Edge min = heap[1];  heap[1] = heap[heapSize--];  adjust(1);  return min;
    }
    static int Find(int i) { while (parent[i] > 0) i = parent[i]; return i; }
    static void Union(int i, int j) { parent[j] = i; }

    static void kruskal() {
        int i = 0, mincost = 0;  buildHeap();
        for (int k=1; k<=n; k++) parent[k] = -1;
        System.out.println("Edges in MST:");
        while (i < n-1 && heapSize > 0) {
            Edge e = deleteMin();
            int j = Find(e.u), k = Find(e.v);
            if (j != k) {
                i++;
                System.out.println(e.u + " -- " + e.v + " : " + e.cost);
                mincost += e.cost;  Union(j, k);
            }
        }
        if (i != n-1) System.out.println("No spanning tree exists");
        else System.out.println("Minimum cost = " + mincost);
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        n = sc.nextInt();  int m = sc.nextInt();
        heap = new Edge[m+1];  parent = new int[n+1];
        System.out.println("Enter edges (u v cost):");
        for (int i=1; i<=m; i++) {
            int u=sc.nextInt(), v=sc.nextInt(), c=sc.nextInt();
            heap[i] = new Edge(u, v, c);
        }
        heapSize = m;  kruskal();  sc.close();
    }
}

Job Sequencing with Deadlines
Select jobs to maximize total profit. Each job has a deadline and takes 1 unit of time. Jobs sorted by profit (decreasing).
Algorithm JS
MA'AM'S ALGORITHM
// p[1]≥p[2]≥...≥p[n].  J[i] = ith job in optimal solution.
JS(d, j, n)
{
  d[0] := J[0] := 0;  J[1] := 1;  k := 1;
  for i := 2 to n do
  {
    r := k;
    while ((d[J[r]] > d[i]) and (d[J[r]] ≠ r))
      do r := r − 1;
    if ((d[J[r]] ≤ d[i]) and (d[i] > r)) then
    {
      for q := k to r+1 step −1 do J[q+1] := J[q];
      J[r+1] := i;  k := k + 1;
    }
  }
  return k;
}

import java.util.Scanner;

class Job {
    int id, deadline, profit;
    Job(int id, int deadline, int profit) {
        this.id=id; this.deadline=deadline; this.profit=profit;
    }
}

public class JobSequencing {

    public static void jobSequencing(Job[] jobs, int n) {
        int[] J = new int[n+1];
        J[0] = 0;  J[1] = 1;  int k = 1;
        for (int i=2; i<=n; i++) {
            int r = k;
            while ((r>0) && (jobs[J[r]].deadline>=jobs[i].deadline)
                        && (jobs[J[r]].deadline!=r))
                r = r - 1;
            if ((jobs[J[r]].deadline <= jobs[i].deadline)
                        && (jobs[i].deadline > r)) {
                for (int q=k; q>=r+1; q--) J[q+1] = J[q];
                J[r+1] = i;  k = k + 1;
            }
        }
        System.out.print("Selected Jobs: ");
        for (int i=1; i<=k; i++) System.out.print("J" + jobs[J[i]].id + " ");
        System.out.println();
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter number of jobs: ");
        int n = sc.nextInt();
        Job[] jobs = new Job[n+1];
        // Enter: id deadline profit (sorted by profit desc)
        System.out.println("Enter job details (id deadline profit):");
        for (int i=1; i<=n; i++)
            jobs[i] = new Job(sc.nextInt(), sc.nextInt(), sc.nextInt());
        jobSequencing(jobs, n);
        sc.close();
    }
}

BST Traversals
Inorder (L→Root→R) gives sorted output. Preorder (Root→L→R) for copying. Postorder (L→R→Root) for deletion.
The Three Traversals
BST: 50 / \ 30 70 / \ 20 40 Inorder: 20 30 40 50 70 ← SORTED! ✓ Preorder: 50 30 20 40 70 ← Root first Postorder: 20 40 30 70 50 ← Root last
Algorithm
PSEUDOCODE
Insert(root, key)
{
  if (root = null) return new Node(key);
  if (key < root.data) root.left := Insert(root.left, key);
  else if (key > root.data) root.right := Insert(root.right, key);
  return root;
}

Inorder(r)   { if r≠null: Inorder(L), visit, Inorder(R) }
Preorder(r)  { if r≠null: visit, Preorder(L), Preorder(R) }
Postorder(r) { if r≠null: Postorder(L), Postorder(R), visit }
Java Code — Insert + All 3 Traversals
Java
import java.util.Scanner;

public class BSTTraversals {

    static class Node {
        int data;  Node left, right;
        Node(int d) { data=d; left=right=null; }
    }

    static Node insert(Node root, int key) {
        if (root == null) return new Node(key);
        if (key < root.data)       root.left  = insert(root.left,  key);
        else if (key > root.data)  root.right = insert(root.right, key);
        return root;
    }

    // Inorder: Left → Root → Right  (gives SORTED output)
    static void inorder(Node r) {
        if (r != null) { inorder(r.left); System.out.print(r.data+" "); inorder(r.right); }
    }

    // Preorder: Root → Left → Right
    static void preorder(Node r) {
        if (r != null) { System.out.print(r.data+" "); preorder(r.left); preorder(r.right); }
    }

    // Postorder: Left → Right → Root
    static void postorder(Node r) {
        if (r != null) { postorder(r.left); postorder(r.right); System.out.print(r.data+" "); }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter number of elements: ");
        int n = sc.nextInt();
        Node root = null;
        System.out.println("Enter elements:");
        for (int i=0; i<n; i++) root = insert(root, sc.nextInt());
        System.out.print("Inorder:   "); inorder(root);   System.out.println();
        System.out.print("Preorder:  "); preorder(root);  System.out.println();
        System.out.print("Postorder: "); postorder(root); System.out.println();
        sc.close();
    }
}
